  <script>
    // JavaScript code for creating birthday wishes
    function createBirthdayWish() {
      var name = document.getElementById('name-input').value;
      var message = document.getElementById('message-input').value;

      if (name && message) {
        var birthdayWishesContainer = document.querySelector('.birthday-wishes-container');
        
        // Create a new birthday wishes card
        var card = document.createElement('div');
        card.className = 'birthday-wishes-card';
        
        // Set the content of the card
        var cardContent = document.createElement('p');
        cardContent.innerHTML = `Happy Birthday, ${name}!<br>${message}`;
        card.appendChild(cardContent);
        
        birthdayWishesContainer.appendChild(card);

        // Clear the text fields after creating the birthday wish
        document.getElementById('name-input').value = '';
        document.getElementById('message-input').value = '';

        // Trigger falling flowers animation
        triggerFallingFlowers();
      } else {
        alert('Please enter a name and message before creating a birthday wish.');
      }
    }

    // JavaScript code for falling flowers animation
    function triggerFallingFlowers() {
      var fallingFlowerContainer = document.querySelector('.falling-flowers');
      
      // Create multiple falling flower elements
      for (var i = 0; i < 10; i++) {
        var flower = document.createElement('div');
        flower.className = 'falling-flower';
        flower.style.left = Math.random() * 100 + 'vw';
        flower.style.animationDuration = Math.random() * 5 + 3 + 's';
        flower.style.animationDelay = Math.random() * 2 + 's';
        fallingFlowerContainer.appendChild(flower);
      }
    }
  </script>
